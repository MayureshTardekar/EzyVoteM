// SPDX-License-Identifier: UNLICENSED
pragma solidity ^0.8.28;
 
contract Voting {
    struct Candidate {
        string name;
        uint256 voteCount;
    }
 
    struct Event {
        string title;
        Candidate[] candidates;
        uint256 endTime;
        bool active;
    }
 
    mapping(address => mapping(uint256 => bool)) public hasVoted;
    mapping(uint256 => Event) public events;
    uint256 public eventCount;
    address public owner;
 
    event VoteCast(
        uint256 indexed eventId,
        address indexed voter,
        uint256 candidateIndex
    );
    event EventCreated(uint256 indexed eventId, string title);
 
    constructor() {
        owner = msg.sender;
    }
 
    modifier onlyOwner() {
        require(msg.sender == owner, "Only owner can call this function");
        _;
    }
 
    function createEvent(
        string memory _title,
        string[] memory _candidateNames,
        uint256 _durationInMinutes
    ) external onlyOwner returns (uint256) {
        eventCount++;
        uint256 eventId = eventCount;
 
        Event storage newEvent = events[eventId];
        newEvent.title = _title;
        newEvent.endTime = block.timestamp + (_durationInMinutes * 1 minutes);
        newEvent.active = true;
 
        for (uint256 i = 0; i < _candidateNames.length; i++) {
            newEvent.candidates.push(
                Candidate({name: _candidateNames[i], voteCount: 0})
            );
        }
 
        emit EventCreated(eventId, _title);
        return eventId;
    }
 
    function vote(uint256 _eventId, uint256 _candidateIndex) external {
        Event storage votingEvent = events[_eventId];
 
        require(votingEvent.active, "Event does not exist or is not active");
        require(block.timestamp < votingEvent.endTime, "Voting has ended");
        require(!hasVoted[msg.sender][_eventId], "Already voted in this event");
        require(
            _candidateIndex < votingEvent.candidates.length,
            "Invalid candidate"
        );
 
        hasVoted[msg.sender][_eventId] = true;
        votingEvent.candidates[_candidateIndex].voteCount++;
 
        emit VoteCast(_eventId, msg.sender, _candidateIndex);
    }
 
    function getEvent(
        uint256 _eventId
    )
        external
        view
        returns (
            string memory title,
            uint256 endTime,
            bool active,
            Candidate[] memory candidates
        )
    {
        Event storage votingEvent = events[_eventId];
        return (
            votingEvent.title,
            votingEvent.endTime,
            votingEvent.active,
            votingEvent.candidates
        );
    }
 
    function endEvent(uint256 _eventId) external onlyOwner {
        require(
            events[_eventId].active,
            "Event does not exist or is already ended"
        );
        events[_eventId].active = false;
    }
}
